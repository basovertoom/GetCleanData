data$date <- as.Date(data$date)

Function to set day name to Weekend and Weekday and add to data_set

for (i in 1:length(day)) {
    if (day[i] == "Saterday") {
        day[i] <- "Weekend"
    } else if (day[i] == "Sunday") {
        day[i] <- "Weekend"
    } else {
        day[i] <- "Weekday"
    }
}

data[, 4] <- day
names(data)[4] <- "day"

Divide data in two sets"weekend" and “weekday” Reshape new dataset for steponinterval for both sets

weekday_set <- subset(data, data$day == "Weekday")
weekend_set <- subset(data, data$day == "Weekend")
